Sliding dots CSS animation

Directive

Directive markup


      <div vlahioSlidingDots [hideMessage]="true" [message]="'Loading...'" [isVisible]="true"></div>

Component markup

Using components directly is highly discouraged. The recommended way of doing it is through dynamic component loading.


      compRef: ComponentRef<SlidingDotsCardComponent>;

      {{'constructor(private slidingDotsWorker: SlidingDotsWorker) {'}}
      {{'}'}}

      {{'show(options: DisplayOptionsInterface): void {'}}
      this.compRef = this.slidingDotsWorker.display(options);
      }

      {{'hide(): void {'}}
      this.slidingDotsWorker.destroy(this.compRef);
      }

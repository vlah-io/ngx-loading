Bouncing dots CSS animation

Directive

Directive markup


      <div vlahioBouncingDots [hideMessage]="true" [message]="'Loading...'" [isVisible]="true"></div>

Component markup

Using components directly is highly discouraged. The recommended way of doing it is through dynamic component loading.


      compRef: ComponentRef<BouncingDotsCardComponent>;

      {{'constructor(private bouncingDotsWorker: BouncingDotsWorker) {'}}
      {{'}'}}

      {{'show(options: DisplayOptionsInterface): void {'}}
      this.compRef = this.bouncingDotsWorker.display(options);
      }

      {{'hide(): void {'}}
      this.bouncingDotsWorker.destroy(this.compRef);
      }
